cv_results <- map_dfr(
  names(vgm_fits),
  ~ {
    model_name <- .x
    fit <- vgm_fits[[.x]]
    
    cv <- krige.cv(log(zinc) ~ 1, meuse, model = fit)
    
    # Extract predictions and observed
    tibble(
      model = model_name,
      observed = cv$observed,
      predicted = cv$var1.pred
    )
  }
)
